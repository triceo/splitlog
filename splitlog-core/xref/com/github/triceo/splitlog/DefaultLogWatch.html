19  /**
20   * Default log watch implementation which provides all the bells and whistles so
21   * that the rest of the tool can work together.
22   * 
23   */
24  final class DefaultLogWatch implements LogWatch {
25  
26      private final Tailer tailer;
27      private final TailSplitter splitter;
28      private final AtomicBoolean isTerminated = new AtomicBoolean(false);
29      private final Set<AbstractLogTailer> tailers = new LinkedHashSet<AbstractLogTailer>();
30      private final LogWatchTailerListener listener;
31      /**
32       * These maps are weak; when a tailer stops being used by user code, we do
33       * not want these IDs to prevent it from being GC'd. Yet, for as long as the
34       * tailer is being used, we want to keep the IDs since the tailer may still
35       * ask for the messages.
36       */
37      private final Map<LogTailer, Integer> startingMessageIds = new WeakHashMap<LogTailer, Integer>(),
38              endingMessageIds = new WeakHashMap<LogTailer, Integer>();
39  
40      private final MessageStore messages;
41      private final MessageCondition acceptanceCondition;
42  
43      protected DefaultLogWatch(final File watchedFile, final TailSplitter splitter, final int capacity,
44              final MessageCondition acceptanceCondition, final long delayBetweenReads,
45              final boolean ignoreExistingContent, final boolean reopenBetweenReads, final int bufferSize) {
46          this.acceptanceCondition = acceptanceCondition;
47          this.messages = new MessageStore(capacity);
48          this.listener = new LogWatchTailerListener(this);
49          this.splitter = splitter;
50          this.tailer = Tailer.create(watchedFile, this.listener, delayBetweenReads, ignoreExistingContent,
51                  reopenBetweenReads, bufferSize);
52      }
53  
54      private MessageBuilder currentlyProcessedMessage = null;
55  
56      protected synchronized void addLine(final String line) {
57          final boolean isMessageBeingProcessed = this.currentlyProcessedMessage != null;
58          if (this.splitter.isStartingLine(line)) {
59              // new message begins
60              if (isMessageBeingProcessed) { // finish old message
61                  this.handleCompleteMessage(this.currentlyProcessedMessage);
62              }
63              // prepare for new message
64              this.currentlyProcessedMessage = new MessageBuilder(line);
65          } else {
66              // continue present message
67              if (!isMessageBeingProcessed) {
68                  // most likely just a garbage immediately after start
69                  return;
70              }
71              this.currentlyProcessedMessage.addLine(line);
72          }
73          this.handleIncomingMessage(this.currentlyProcessedMessage);
74      }
75  
76      private synchronized void handleIncomingMessage(final MessageBuilder messageBuilder) {
77          final Message message = messageBuilder.buildIntermediate(this.splitter);
78          for (final AbstractLogTailer t : this.tailers) {
79              t.notifyOfIncomingMessage(message);
80          }
81      }
82  
83      private synchronized void handleUndeliveredMessage(final MessageBuilder messageBuilder) {
84          final Message message = messageBuilder.buildIntermediate(this.splitter);
85          for (final AbstractLogTailer t : this.tailers) {
86              t.notifyOfUndeliveredMessage(message);
87          }
88      }
89  
90      private synchronized void handleCompleteMessage(final MessageBuilder messageBuilder) {
91          final Message message = messageBuilder.buildFinal(this.splitter);
92          final boolean messageAccepted = this.acceptanceCondition.accept(message);
93          if (messageAccepted) {
94              this.messages.add(message);
95          }
96          for (final AbstractLogTailer t : this.tailers) {
97              if (messageAccepted) {
98                  t.notifyOfAcceptedMessage(message);
99              } else {
100                 t.notifyOfRejectedMessage(message);
101             }
102         }
103     }

105     /**
106      * Return all messages that have been sent to the tailer, from its start
107      * until either its termination or to this moment, whichever is relevant.
108      * 
109      * This method is synchronized so that the modification of the underlying
110      * message store in {@link #addLine(String)} and the reading of this store
111      * is mutually excluded. Otherwise, there is a possibility of message ID
112      * mess in the discarding case.
113      * 
114      * @param tail
115      *            The tailer in question.
116      * @return Unmodifiable map of all the received messages, with keys being
117      *         IDs of those messages.
118      */
119     protected synchronized SortedMap<Integer, Message> getAllMessages(final LogTailer tail) {
120         final int start = this.getStartingMessageId(tail);
121         // get the expected ending message ID
122         final int end = this.getEndingMessageId(tail);
123         if (start > end) {
124             /*
125              * in case some messages have been discarded, the actual start may
126              * get ahead of the expected end. this would have caused an
127              * exception within the message store, and so we handle it here and
128              * return an empty list. this is exactly correct, as if the end is
129              * before the first message in the store, there really is nothing to
130              * return.
131              */
132             return Collections.unmodifiableSortedMap(new TreeMap<Integer, Message>());
133         } else {
134             final SortedMap<Integer, Message> messages = new TreeMap<Integer, Message>();
135             int id = start;
136             for (final Message msg : this.messages.getFromRange(start, end + 1)) {
137                 messages.put(id, msg);
138                 id++;
139             }
140             return Collections.unmodifiableSortedMap(messages);
141         }
142     }

144     /**
145      * Get index of the last plus one message that the tailer has access to.
146      * 
147      * @param tail
148      *            Tailer in question.
149      */
150     protected int getEndingMessageId(final LogTailer tail) {
151         return this.endingMessageIds.containsKey(tail) ? this.endingMessageIds.get(tail) : this.messages
152                 .getLatestMessageId();
153     }
154 
155     /**
156      * If messages have been discarded, the original starting message ID will no
157      * longer be valid. therefore, we check for the actual starting ID.
158      * 
159      * @param tail
160      *            Tailer in question.
161      */
162     protected int getStartingMessageId(final LogTailer tail) {
163         return Math.max(this.messages.getFirstMessageId(), this.startingMessageIds.get(tail));
164     }

166     @Override
167     public boolean isTerminated() {
168         return this.isTerminated.get();
169     }
170 
171     @Override
172     public boolean isTerminated(final LogTailer tail) {
173         return !this.tailers.contains(tail);
174     }

176     @Override
177     public synchronized LogTailer startTailing() {
178         if (this.isTerminated()) {
179             throw new IllegalStateException("Cannot start tailing on an already terminated LogWatch.");
180         }
181         final int startingMessageId = this.messages.getNextMessageId();
182         final AbstractLogTailer tail = new NonStoringLogTailer(this);
183         this.tailers.add(tail);
184         this.startingMessageIds.put(tail, startingMessageId);
185         return tail;
186     }
187 
188     @Override
189     public synchronized boolean terminateTailing() {
190         if (this.isTerminated()) {
191             return false;
192         }
193         this.tailer.stop();
194         this.isTerminated.set(true);
195         if (this.currentlyProcessedMessage != null) {
196             this.handleUndeliveredMessage(this.currentlyProcessedMessage);
197         }
198         for (final AbstractLogTailer chunk : new ArrayList<AbstractLogTailer>(this.tailers)) {
199             this.terminateTailing(chunk);
200         }
201         return true;
202     }
203 
204     @Override
205     public synchronized boolean terminateTailing(final LogTailer tail) {
206         if (this.tailers.remove(tail)) {
207             final int endingMessageId = this.messages.getLatestMessageId();
208             this.endingMessageIds.put(tail, endingMessageId);
209             return true;
210         } else {
211             return false;
212         }
213     }
