3   import java.util.Collection;

5   /**
6    * Follower that is capable of merging multiple {@link Follower}s.
7    *
8    * They receive all messages that their {@link #getMerged()} receive. If more
9    * followers receive the same message, this receive all of them.
10   *
11   * Unlike {@link Follower}, this one can not tag. However, it will retrieve
12   * every tag that has been made using any of the {@link #getMerged()}.
13   *
14   */
15  public interface MergingFollower extends CommonFollower<MergingFollower, Follower> {
16  
17      /**
18       * Retrieve followers that are currently part of this merge.
19       *
20       * @return Unmodifiable collections of followers in this merge.
21       */
22      Collection<Follower> getMerged();
23  
24      /**
25       * Will return an instance whose {@link #getMerged()} does not contain the
26       * given follower.
27       *
28       * @param f
29       *            The follower to remove from the merge.
30       * @return New instance of the follower containing the followers without
31       *         this one, if {@link #getMerged()} contained it. If it didn't, the
32       *         same instance is returned. Null is returned when the merge would
33       *         be empty after this call.
34       */
35      MergingFollower remove(Follower f);
36  
37  }
