11  import com.github.triceo.splitlog.api.Follower;
12  import com.github.triceo.splitlog.api.LogWatchBuilder;
13  import com.github.triceo.splitlog.api.Message;
14  import com.github.triceo.splitlog.api.SimpleMessageCondition;
15  
16  @RunWith(Parameterized.class)
17  public class GetMessagesTest extends DefaultFollowerBaseTest {
18  
19      private static final class NumberEndingMessageCondition implements SimpleMessageCondition {
20  
21          @Override
22          public boolean accept(final Message evaluate) {
23              final String line = evaluate.getLines().get(0);
24              final char endingCharacter = line.charAt(line.length() - 1);
25              return Character.isDigit(endingCharacter);
26          }
27  
28      }
29  
30      private static final class TestStartingMessageCondition implements SimpleMessageCondition {
31  
32          @Override
33          public boolean accept(final Message evaluate) {
34              return (evaluate.getLines().get(0).startsWith("test"));
35          }
36  
37      }
38  
39      public GetMessagesTest(final LogWatchBuilder builder) {
40          super(builder);
41      }
42  
43      @Test
44      public void testGetMessages() {
45          final String message1 = "test";
46          final String message2part1 = "test1";
47          final String message2part2 = "test2";
48          final String message3part1 = "test3";
49          final String message3part2 = "test4";
50          final Follower follower = this.getLogWatch().startFollowing();
51          // test simple messages
52          String result = LogWriter.write(follower, message1);
53          Assertions.assertThat(result).isEqualTo(message1);
54          result = LogWriter.write(follower, message1);
55          Assertions.assertThat(result).isEqualTo(message1);
56          result = LogWriter.write(follower, message2part1 + "\n" + message2part2);
57          Assertions.assertThat(result).isEqualTo(message2part2);
58          result = LogWriter.write(follower, message3part1 + "\r\n" + message3part2);
59          Assertions.assertThat(result).isEqualTo(message3part2);
60          // now validate the results with the default condition
61          List<Message> messages = new LinkedList<Message>(follower.getMessages());
62          Assertions.assertThat(messages.size()).isEqualTo(5);
63          Assertions.assertThat(messages.get(1).getLines().get(0)).isEqualTo(message1);
64          Assertions.assertThat(messages.get(2).getLines().get(0)).isEqualTo(message2part1);
65          Assertions.assertThat(messages.get(3).getLines().get(0)).isEqualTo(message2part2);
66          Assertions.assertThat(messages.get(4).getLines().get(0)).isEqualTo(message3part1);
67          // now validate a condition that will accept all messages
68          messages = new LinkedList<Message>(follower.getMessages(new TestStartingMessageCondition()));
69          Assertions.assertThat(messages.size()).isEqualTo(5);
70          Assertions.assertThat(messages.get(1).getLines().get(0)).isEqualTo(message1);
71          Assertions.assertThat(messages.get(2).getLines().get(0)).isEqualTo(message2part1);
72          Assertions.assertThat(messages.get(3).getLines().get(0)).isEqualTo(message2part2);
73          Assertions.assertThat(messages.get(4).getLines().get(0)).isEqualTo(message3part1);
74          // now validate a condition that will only accept messages ending with a
75          // numeric digit
76          messages = new LinkedList<Message>(follower.getMessages(new NumberEndingMessageCondition()));
77          Assertions.assertThat(messages.size()).isEqualTo(3);
78          Assertions.assertThat(messages.get(0).getLines().get(0)).isEqualTo(message2part1);
79          Assertions.assertThat(messages.get(1).getLines().get(0)).isEqualTo(message2part2);
80          Assertions.assertThat(messages.get(2).getLines().get(0)).isEqualTo(message3part1);
81          // final part of the message, message3part2, will remain unflushed
82          this.getLogWatch().stopFollowing(follower);
83          Assertions.assertThat(follower.isStopped()).isTrue();
84      }
