5   import org.assertj.core.api.Assertions;
6   import org.junit.After;

12  import com.github.triceo.splitlog.api.Follower;
13  import com.github.triceo.splitlog.api.LogWatchBuilder;
14  import com.github.triceo.splitlog.api.Message;
15  import com.github.triceo.splitlog.logging.SplitlogLoggerFactory;
16  
17  @RunWith(Parameterized.class)
18  public class ReferenceBasedDiscardingTest extends DefaultFollowerBaseTest {
19  
20      public ReferenceBasedDiscardingTest(final LogWatchBuilder builder) {
21          super(builder.withDelayBetweenSweeps(1, TimeUnit.SECONDS));
22      }
23  
24      @Before
25      public void disableLogging() {
26          /*
27           * logging, when async, may store messages and their attributes for
28           * future writing. this will keep the followers from being GC'd and will
29           * break this test.
30           */
31          SplitlogLoggerFactory.silenceLogging();
32      }
33  
34      @After
35      public void enableLogging() {
36          SplitlogLoggerFactory.resetLoggingToDefaultState();
37      }
38  
39      @Test
40      public void test() {
41          final DefaultLogWatch w = (DefaultLogWatch) this.getLogWatch();
42          Follower follower = w.startFollowing();
43          // tag before any messages
44          final Message firstTag = follower.tag("test");
45          DefaultFollowerBaseTest.assertProperOrder(follower.getMessages(), firstTag);
46          final String firstMessage = "check";
47          LogWriter.write(follower, firstMessage);
48          // receive first message, check presence of tag
49          final String secondMessage = "check2";
50          LogWriter.write(follower, secondMessage);
51          final Message secondTag = follower.tag("test2");
52          DefaultFollowerBaseTest.assertProperOrder(follower.getMessages(), firstTag, firstMessage, secondTag);
53          // start second follower; this one will only track second+ messages
54          Follower follower2 = w.startFollowing();
55          // send third message, receive second
56          final String thirdMessage = "check3";
57          LogWriter.write(follower2, thirdMessage);
58          Assertions.assertThat(w.countMessagesInStorage()).isEqualTo(2);
59          /*
60           * remove all references to the first follower; the first message now
61           * has no followers available and can be GC'd
62           */
63          w.stopFollowing(follower);
64          follower = null;
65          System.gc();
66          final long delay = this.getBuilder().getDelayBetweenSweeps() + 500;
67          try {
68              Thread.sleep(delay);
69          } catch (final InterruptedException e) {
70              Assertions.fail("Test will fail as there was not enough time to wait for message sweep.");
71          }
72          Assertions.assertThat(w.countMessagesInStorage()).isEqualTo(1);
73          DefaultFollowerBaseTest.assertProperOrder(follower2.getMessages(), secondMessage);
74          // terminate following, make sure all the messages are cleared
75          w.stopFollowing(follower2);
76          follower2 = null;
77          System.gc();
78          try {
79              Thread.sleep(delay);
80          } catch (final InterruptedException e) {
81              Assertions.fail("Test will fail as there was not enough time to wait for message sweep.");
82          }
83          Assertions.assertThat(w.countMessagesInStorage()).isEqualTo(0);
84      }
