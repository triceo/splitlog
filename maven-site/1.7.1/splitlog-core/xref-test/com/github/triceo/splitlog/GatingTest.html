27      private static final String MEASURE_ID = "test";
28  
29      // will verify various configs of log watch
30      @Parameters(name = "{index}: {0}, {1}")
31      public static Collection<Object[]> data() {
32          return Arrays.asList(new Object[][] {
33                  { LogWatchBuilder.getDefault().watchedFile(LogWriter.createTempFile()), false },
34                  {
35                          LogWatchBuilder.getDefault().watchedFile(LogWriter.createTempFile())
36                                  .withGateCondition(SplitlogMessagesRejectingCondition.INSTANCE), false },
37                  {
38                          LogWatchBuilder.getDefault().watchedFile(LogWriter.createTempFile())
39                                  .withGateCondition(AllFollowerMessagesAcceptingCondition.INSTANCE), true } });
40      }
41  
42      private final boolean isGatingDisabled;
43  
44      public GatingTest(final LogWatchBuilder builder, final boolean gatingDisabled) {
45          super(builder);
46          this.isGatingDisabled = gatingDisabled;
47      }
48  
49      private MessageMetric<Integer, LogWatch> fillWithData(final DefaultLogWatch watch, final TailSplitter splitter) {
50          final MessageMetric<Integer, LogWatch> metric = watch.startMeasuring(new MessageMeasure<Integer, LogWatch>() {
51  
52              @Override
53              public Integer initialValue() {
54                  return 0;
55              }
56  
57              @Override
58              public Integer update(final MessageMetric<Integer, LogWatch> metric, final Message evaluate,
59                  final MessageDeliveryStatus status, final LogWatch source) {
60                  if (status != MessageDeliveryStatus.INCOMING) {
61                      return metric.getValue() + 1;
62                  } else {
63                      return metric.getValue();
64                  }
65              }
66  
67          }, GatingTest.MEASURE_ID);
68          watch.messageArrived(this.getMessage(
69                  "07:30:02,670 INFO  [org.jboss.msc] (main) JBoss MSC version 1.0.4.GA-redhat-1", splitter));
70          watch.messageArrived(this.getMessage("07:30:02,670 INFO  [com.github.triceo.splitlog] (main) random", splitter));
71          watch.messageArrived(this.getMessage("07:30:02,670 INFO  [com.github.triceo.splitlog.check] (main) random 2",
72                  splitter));
73          watch.messageArrived(this.getMessage(
74                  "07:30:02,739 DEBUG [org.jboss.as.config] (MSC service thread 1-7) Configured system properties:",
75                  splitter));
76          watch.messageIncoming(this
77                  .getMessage(
78                          "07:30:02,731 INFO  [org.jboss.as] (MSC service thread 1-7) JBAS015899: JBoss BRMS 6.0.1.GA (AS 7.2.1.Final-redhat-10) starting",
79                          splitter));
80          watch.stop();
81          return metric;
82      }
83  
84      private Message getMessage(final String line, final TailSplitter splitter) {
85          return new MessageBuilder(line).buildFinal(splitter);
86      }
87  
88      private TailSplitter getProperSplitter() {
89          return (this.isGatingDisabled) ? new SimpleTailSplitter() : new JBossServerLogTailSplitter();
90      }
91  
92      @Test
93      public void testDefaults() {
94          // with simple tail splitter, no gatting can take place
95          final DefaultLogWatch watch = (DefaultLogWatch) this.getLogWatch();
96          final MessageMetric<Integer, LogWatch> metric = this.fillWithData(watch, new SimpleTailSplitter());
97          Assertions.assertThat(metric.getValue()).isEqualTo(4);
98      }
99  
100     @Test
101     public void testGatingCondition() {
102 
103         final DefaultLogWatch watch = (DefaultLogWatch) this.getBuilder().buildWith(new JBossServerLogTailSplitter());
104         final MessageMetric<Integer, LogWatch> metric = this.fillWithData(watch, this.getProperSplitter());
105         if (this.isGatingDisabled) {
106             // all messages will be let through
107             Assertions.assertThat(metric.getValue()).isEqualTo(4);
108         } else {
109             // only messages not from splitlog will be let through
110             Assertions.assertThat(metric.getValue()).isEqualTo(2);
111         }
112     }
