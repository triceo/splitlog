1   package com.github.triceo.splitlog.formatters;
2   
3   import java.io.File;
4   
5   import com.github.triceo.splitlog.api.Message;
6   import com.github.triceo.splitlog.api.MessageFormatter;
7   
8   /**
9    * This formatter will output the message in a format as similar to the original
10   * as possible. Some whitespace may still be mangled and line endings will be in
11   * the format appropriate for the current environment.
12   */
13  public final class NoopMessageFormatter implements MessageFormatter {
14  
15      public static final MessageFormatter INSTANCE = new NoopMessageFormatter();
16  
17      private NoopMessageFormatter() {
18          // singleton
19      }
20  
21      @Override
22      public String format(final Message m, final File source) {
23          final StringBuilder sb = new StringBuilder();
24          for (final String line : m.getLines()) {
25              sb.append(line);
26              sb.append(MessageFormatter.LINE_SEPARATOR);
27          }
28          return sb.toString().trim();
29      }
30  }
