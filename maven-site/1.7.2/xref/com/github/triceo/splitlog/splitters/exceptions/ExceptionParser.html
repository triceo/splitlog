5   /**
6    * A finite automaton for recognizing a Java exception stack trace spanning
7    * multiple lines in a random text.
8    *
9    * Each line has a type. When a line is recognized, a new state is reached. Each
10   * line type can only be followed by lines of specific types, together forming a
11   * transition function. When the next line is not one of these types, that
12   * transition is not in the transition function and the parsing is terminated
13   * with an exception. After the last line has been read, if the automaton is not
14   * in one of the accepting states, the parsing is terminated with an exception.
15   */
16  final class ExceptionParser {
17  
18      private final List<ExceptionLine> parsedLines = new LinkedList<ExceptionLine>();
19  
20      private static String greatestCommonPrefix(final String a, final String b) {
21          final int minLength = Math.min(a.length(), b.length());
22          for (int i = 0; i < minLength; i++) {
23              if (a.charAt(i) != b.charAt(i)) {
24                  return a.substring(0, i);
25              }
26          }
27          return a.substring(0, minLength);
28      }
29  
30      /**
31       * Identifies and removes the common prefix - the longest beginning
32       * substring that all the lines share.
33       *
34       * @param input
35       *            Lines to be evaluated.
36       * @return The same lines, without the prefix. And stripped of white space
37       *         at the ends.
38       */
39      private static Queue<String> removePrefix(final List<String> input) {
40          if (input.size() < 2) {
41              return new LinkedList<String>(input);
42          }
43          String resultingPrefix = "";
44          final String previousGreatestCommonPrefix = input.get(0).trim();
45          String greatestCommonPrefix = "";
46          for (int i = 1; i < input.size(); i++) {
47              final String previousLine = input.get(i - 1).trim();
48              final String currentLine = input.get(i).trim();
49              greatestCommonPrefix = ExceptionParser.greatestCommonPrefix(previousLine, currentLine);
50              resultingPrefix = ExceptionParser.greatestCommonPrefix(previousGreatestCommonPrefix, greatestCommonPrefix);
51              if (resultingPrefix.length() == 0) {
52                  break;
53              }
54          }
55          final int prefixLength = resultingPrefix.length();
56          final boolean hasPrefix = prefixLength > 0;
57          final Queue<String> result = new LinkedList<String>();
58          for (final String line : input) {
59              final String line2 = line.trim();
60              if (hasPrefix) {
61                  result.add(line2.substring(prefixLength));
62              } else {
63                  result.add(line2);
64              }
65          }
66          return result;
67      }

69      /**
70       * Browsers through a random log and returns first exception stack trace it
71       * could find.
72       *
73       * @param input Lines of the log. May begin and end with any garbage, may or
74       *              may not contain exception.
75       * @return Raw exception data.
76       * @throws ExceptionParseException If parsing of the log file failed.
77       */
78      public synchronized Collection<ExceptionLine> parse(final Collection<String> input) throws ExceptionParseException {
79          this.parsedLines.clear();
80          final Queue<String> linesFromInput = ExceptionParser.removePrefix(new LinkedList<String>(input));
81          LineType previousLineType = LineType.PRE_START;
82          String currentLine = null;
83          boolean isFirstLine = true;
84          while (!linesFromInput.isEmpty()) {
85              currentLine = linesFromInput.poll();
86              previousLineType = this.parseLine(previousLineType, currentLine);
87              if (isFirstLine) {
88                  if (!previousLineType.isAcceptableAsFirstLine()) {
89                      throw new ExceptionParseException(currentLine, "Invalid line type detected at the beginning: "
90                              + previousLineType);
91                  }
92                  isFirstLine = false;
93              }
94          }
95          if (!previousLineType.isAcceptableAsLastLine()) {
96              throw new ExceptionParseException(currentLine, "Invalid line type detected at the end: " + previousLineType);
97          }
98          return Collections.unmodifiableCollection(new LinkedList<ExceptionLine>(this.parsedLines));
99      }

101     /**
102      * Parse one line in the log.
103      *
104      * @param previousLineType
105      *            Type of the previous line in the log. (The state the automaton
106      *            is currently in.)
107      * @param line
108      *            Line in question.
109      * @return Identified type of this line.
110      * @throws ExceptionParseException
111      *             If parsing of the log file failed.
112      */
113     private LineType parseLine(final LineType previousLineType, final String line) throws ExceptionParseException {
114         switch (previousLineType) {
115             case PRE_START:
116                 return this.parseLine(line, LineType.CAUSE, LineType.PRE_START);
117             case CAUSE:
118             case SUB_CAUSE:
119             case PRE_STACK_TRACE:
120                 return this.parseLine(line, LineType.STACK_TRACE, LineType.PRE_STACK_TRACE);
121             case STACK_TRACE:
122                 return this.parseLine(line, LineType.STACK_TRACE, LineType.STACK_TRACE_END, LineType.SUB_CAUSE);
123             case STACK_TRACE_END:
124                 return this.parseLine(line, LineType.SUB_CAUSE, LineType.POST_END);
125             case POST_END:
126                 return this.parseLine(line, LineType.POST_END);
127             default:
128                 throw new IllegalArgumentException("Unsupported line type: " + previousLineType);
129         }
130     }
131 
132     /**
133      * Parse one line in the log, when knowing the types of lines acceptable at
134      * this point in the log.
135      *
136      * @param line
137      *            Line to parse.
138      * @param allowedLineTypes
139      *            Possible line types, in the order of evaluation. (Possible
140      *            transitions.) If evaluation matches for a type, transition
141      *            will be made and any subsequent types will be ignored.
142      * @return Identified type of this line.
143      * @throws ExceptionParseException
144      *             If no allowed types match.
145      */
146     private LineType parseLine(final String line, final LineType... allowedLineTypes) throws ExceptionParseException {
147         for (final LineType possibleType : allowedLineTypes) {
148             if ((possibleType == LineType.POST_END) || (possibleType == LineType.PRE_START)) {
149                 // this is garbage, all is accepted without parsing
150                 return possibleType;
151             }
152             final ExceptionLine parsedLine = possibleType.parse(line);
153             if (parsedLine != null) {
154                 if (possibleType == LineType.PRE_STACK_TRACE) {
155                     // in case the cause is multi-line, update the cause
156                     final ExceptionLine currentTop = this.parsedLines.remove(this.parsedLines.size() - 1);
157                     if (!((currentTop instanceof CauseLine) && (parsedLine instanceof PlainTextLine))) {
158                         throw new IllegalStateException("Garbage in the exception message.");
159                     }
160                     this.parsedLines.add(new CauseLine((CauseLine) currentTop, (PlainTextLine) parsedLine));
161                 } else {
162                     this.parsedLines.add(parsedLine);
163                 }
164                 return possibleType;
165             }
166         }
167         throw new ExceptionParseException(line, "Line not any of the expected types: "
168                 + Arrays.toString(allowedLineTypes));
169     }
170 
171     /**
172      * Various kinds of states for the parser automaton.
173      */
174     private enum LineType {
175         CAUSE(true, false), POST_END(false, true), PRE_START(true, false), PRE_STACK_TRACE(false, false),
176         STACK_TRACE(false, true), STACK_TRACE_END(false, true), SUB_CAUSE(false, false);
177 
178         private final boolean mayBeFirstLine, mayBeLastLine;
179         private ExceptionLineParser<?> parser;
180 
181         LineType(final boolean mayBeginWith, final boolean mayEndWith) {
182             this.mayBeFirstLine = mayBeginWith;
183             this.mayBeLastLine = mayEndWith;
184         }
185 
186         private ExceptionLineParser<?> determineParser() {
187             switch (this) {
188                 case PRE_START:
189                 case POST_END:
190                     return null;
191                 case CAUSE:
192                     return new CauseParser();
193                 case PRE_STACK_TRACE:
194                     return new CopyingParser();
195                 case STACK_TRACE:
196                     return new StackTraceParser();
197                 case STACK_TRACE_END:
198                     return new StackTraceEndParser();
199                 case SUB_CAUSE:
200                     return new SubCauseParser();
201             }
202             throw new IllegalStateException("This can never happen.");
203         }
204 
205         /**
206          * Whether or not this type can be a start state.
207          */
208         public boolean isAcceptableAsFirstLine() {
209             return this.mayBeFirstLine;
210         }
211 
212         /**
213          * Whether or not this type can be an accepting state.
214          */
215         public boolean isAcceptableAsLastLine() {
216             return this.mayBeLastLine;
217         }
218 
219         /**
220          * Parse the line according to this state's parser.
221          *
222          * @param line Line to parse.
223          * @return Parsed line.
224          * @throws ExceptionParseException If the parser doesn't recognize the line.
225          */
226         public ExceptionLine parse(final String line) throws ExceptionParseException {
227             if ((this == LineType.POST_END) || (this == LineType.PRE_START)) {
228                 throw new IllegalStateException("No need to parse garbage lines.");
229             } else if (this.parser == null) {
230                 this.parser = this.determineParser();
231             }
232             return this.parser.parse(line);
233         }
234     }
