3   import java.io.File;
4   
5   import com.github.triceo.splitlog.api.Message;
6   import com.github.triceo.splitlog.api.MessageFormatter;
7   
8   /**
9    * This formatter will output the message in a format specific to Splitlog. It
10   * will strip metadata out of the original message and only include some of them
11   * in the new output.
12   */
13  public final class UnifyingMessageFormatter implements MessageFormatter {
14  
15      public static final MessageFormatter INSTANCE = new UnifyingMessageFormatter();
16  
17      private UnifyingMessageFormatter() {
18          // singleton
19      }
20  
21      @Override
22      public String format(final Message m, final File source) {
23          final StringBuilder sb = new StringBuilder();
24          sb.append("[");
25          sb.append(source);
26          sb.append("] #");
27          sb.append(m.getUniqueId());
28          sb.append(" ");
29          sb.append(m.getDate());
30          sb.append(" (");
31          sb.append(m.getType());
32          sb.append(") ");
33          sb.append(m.getSeverity());
34          sb.append(" ");
35          for (final String line : m.getLinesWithoutMetadata()) {
36              sb.append(line);
37              sb.append(MessageFormatter.LINE_SEPARATOR);
38          }
39          return sb.toString().trim();
40      }
41  }
