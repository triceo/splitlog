1   package com.github.triceo.splitlog;

3   import java.util.concurrent.TimeUnit;
4   
5   import org.junit.Rule;
6   import org.junit.rules.TestRule;
7   import org.junit.rules.TestWatcher;
8   import org.junit.rules.Timeout;
9   import org.junit.runner.Description;
10  import org.slf4j.Logger;
11  import org.slf4j.LoggerFactory;
12  
13  /**
14   * Base class for all Splitlog tests. Will enforce a default timeout and add
15   * some useful test execution logging.
16   */
17  public abstract class AbstractSplitlogTest {
18  
19      private static final Logger LOGGER = LoggerFactory.getLogger(AbstractSplitlogTest.class);
20  
21      @Rule
22      public Timeout globalTimeout = new Timeout((int) TimeUnit.MINUTES.toMillis(1));
23  
24      @Rule
25      public TestRule watchman = new TestWatcher() {
26  
27          @Override
28          protected void finished(final Description d) {
29              AbstractSplitlogTest.LOGGER.info("----- Finished test: {}\n", d);
30          }
31  
32          @Override
33          protected void starting(final Description d) {
34              AbstractSplitlogTest.LOGGER.info("----- Starting test: {}", d);
35          }
36      };
37  
38  }
