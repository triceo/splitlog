58      private static class TestTailerListener implements TailerListener {
59  
60          volatile Exception exception = null;
61  
62          volatile int initialised = 0;
63  
64          // Must be synchronised because it is written by one thread and read by another
65          private final List<String> lines = Collections.synchronizedList(new ArrayList<String>());
66  
67          volatile int notFound = 0;
68  
69          volatile int rotated = 0;
70  
71          @Override
72          public void begin() {
73              // do nothing
74          }
75  
76          public void clear() {
77              this.lines.clear();
78          }
79  
80          @Override
81          public void commit() {
82              // do nothing
83          }
84  
85          @Override
86          public void destroy() {
87              // do nothing
88          }
89  
90          @Override
91          public void fileNotFound() {
92              this.notFound++; // not atomic, but OK because only updated here.
93          }
94  
95          @Override
96          public void fileRotated() {
97              this.rotated++; // not atomic, but OK because only updated here.
98          }
99  
100         public List<String> getLines() {
101             return this.lines;
102         }
103 
104         @Override
105         public void handle(final Exception e) {
106             this.exception = e;
107         }
108 
109         @Override
110         public void handle(final String line) {
111             this.lines.add(line);
112         }
113 
114         @Override
115         public void init(final Tailer tailer) {
116             this.initialised++; // not atomic, but OK because only updated here.
117         }
118     }

126     @Override
127     protected void createFile(final File file, final long size)
128             throws IOException {
129         super.createFile(file, size);
130 
131         // try to make sure file is found
132         // (to stop continuum occasionally failing)
133         RandomAccessFile reader = null;
134         try {
135             while (reader == null) {
136                 try {
137                     reader = new RandomAccessFile(file.getPath(), "r");
138                 } catch (final FileNotFoundException e) {
139                 }
140                 try {
141                     Thread.sleep(200L);
142                 } catch (final InterruptedException e) {
143                     // ignore
144                 }
145             }
146         } finally {
147             IOUtils.closeQuietly(reader);
148         }
149     }

182     /**
183      * Tests [IO-357][Tailer] InterruptedException while the thead is sleeping is silently ignored.
184      *
185      * @throws Exception
186      */
187     public void testInterrupt() throws Exception {
188         final File file = new File(FileBasedTestCase.getTestDirectory(), "nosuchfile");
189         TestCase.assertFalse("nosuchfile should not exist", file.exists());
190         final TestTailerListener listener = new TestTailerListener();
191         // Use a long delay to try to make sure the test thread calls interrupt() while the tailer thread is sleeping.
192         final int delay = 1000;
193         final int idle = 50; // allow time for thread to work
194         Tailer tailer = new Tailer(file, listener, delay, false, 4096);
195         final Thread thread = new Thread(tailer);
196         thread.setDaemon(true);
197         thread.start();
198         Thread.sleep(idle);
199         thread.interrupt();
200         tailer = null;
201         Thread.sleep(delay + idle);
202         TestCase.assertNotNull("Missing InterruptedException", listener.exception);
203         TestCase.assertTrue("Unexpected Exception: " + listener.exception, listener.exception instanceof InterruptedException);
204         TestCase.assertEquals("Expected init to be called", 1, listener.initialised);
205         TestCase.assertTrue("fileNotFound should be called", listener.notFound > 0);
206         TestCase.assertEquals("fileRotated should be not be called", 0, listener.rotated);
207     }

266     public void testMultiByteBreak() throws Exception {
267         System.out.println("testMultiByteBreak() Default charset: "+Charset.defaultCharset().displayName());
268         final long delay = 50;
269         final File origin = new File(this.getClass().getResource("/test-file-utf8.bin").toURI());
270         final File file = new File(FileBasedTestCase.getTestDirectory(), "testMultiByteBreak.txt");
271         this.createFile(file, 0);
272         final TestTailerListener listener = new TestTailerListener();
273         final String osname = System.getProperty("os.name");
274         final boolean isWindows = osname.startsWith("Windows");
275         // Need to use UTF-8 to read & write the file otherwise it can be corrupted (depending on the default charset)
276         final Charset charsetUTF8 = Charsets.UTF_8;
277         this.tailer = new Tailer(file, charsetUTF8, listener, delay, false, isWindows, 4096);
278         final Thread thread = new Thread(this.tailer);
279         thread.start();
280 
281         final Writer out = new OutputStreamWriter(new FileOutputStream(file), charsetUTF8);
282         BufferedReader reader = null;
283         try{
284             final List<String> lines = new ArrayList<String>();
285             reader = new BufferedReader(new InputStreamReader(new FileInputStream(origin), charsetUTF8));
286             String line = null;
287             while((line = reader.readLine()) != null){
288                 out.write(line);
289                 out.write("\n");
290                 lines.add(line);
291             }
292             out.close(); // ensure data is written
293 
294             final long testDelayMillis = delay * 10;
295             Thread.sleep(testDelayMillis);
296             final List<String> tailerlines = listener.getLines();
297             TestCase.assertEquals("line count",lines.size(),tailerlines.size());
298             for(int i = 0,len = lines.size();i<len;i++){
299                 final String expected = lines.get(i);
300                 final String actual = tailerlines.get(i);
301                 if (!expected.equals(actual)) {
302                     TestCase.fail("Line: " + i
303                             + "\nExp: (" + expected.length() + ") " + expected
304                             + "\nAct: (" + actual.length() + ") "+ actual);
305                 }
306             }
307         }finally{
308             this.tailer.stop();
309             IOUtils.closeQuietly(reader);
310             IOUtils.closeQuietly(out);
311         }
312     }

350     public void testTailer() throws Exception {
351 
352         // Create & start the Tailer
353         final long delayMillis = 50;
354         final File file = new File(FileBasedTestCase.getTestDirectory(), "tailer1-test.txt");
355         this.createFile(file, 0);
356         final TestTailerListener listener = new TestTailerListener();
357         final String osname = System.getProperty("os.name");
358         final boolean isWindows = osname.startsWith("Windows");
359         this.tailer = new Tailer(file, listener, delayMillis, false, isWindows);
360         final Thread thread = new Thread(this.tailer);
361         thread.start();
362 
363         // Write some lines to the file
364         this.write(file, "Line one", "Line two");
365         final long testDelayMillis = delayMillis * 10;
366         Thread.sleep(testDelayMillis);
367         List<String> lines = listener.getLines();
368         TestCase.assertEquals("1 line count", 2, lines.size());
369         TestCase.assertEquals("1 line 1", "Line one", lines.get(0));
370         TestCase.assertEquals("1 line 2", "Line two", lines.get(1));
371         listener.clear();
372 
373         // Write another line to the file
374         this.write(file, "Line three");
375         Thread.sleep(testDelayMillis);
376         lines = listener.getLines();
377         TestCase.assertEquals("2 line count", 1, lines.size());
378         TestCase.assertEquals("2 line 3", "Line three", lines.get(0));
379         listener.clear();
380 
381         // Check file does actually have all the lines
382         lines = FileUtils.readLines(file, "UTF-8");
383         TestCase.assertEquals("3 line count", 3, lines.size());
384         TestCase.assertEquals("3 line 1", "Line one", lines.get(0));
385         TestCase.assertEquals("3 line 2", "Line two", lines.get(1));
386         TestCase.assertEquals("3 line 3", "Line three", lines.get(2));
387 
388         // Delete & re-create
389         file.delete();
390         final boolean exists = file.exists();
391         TestCase.assertFalse("File should not exist", exists);
392         this.createFile(file, 0);
393         Thread.sleep(testDelayMillis);
394 
395         // Write another line
396         this.write(file, "Line four");
397         Thread.sleep(testDelayMillis);
398         lines = listener.getLines();
399         TestCase.assertEquals("4 line count", 1, lines.size());
400         TestCase.assertEquals("4 line 3", "Line four", lines.get(0));
401         listener.clear();
402 
403         // Stop
404         this.tailer.stop();
405         this.tailer=null;
406         thread.interrupt();
407         Thread.sleep(testDelayMillis * 4);
408         this.write(file, "Line five");
409         TestCase.assertEquals("4 line count", 0, listener.getLines().size());
410         TestCase.assertNotNull("Missing InterruptedException", listener.exception);
411         TestCase.assertTrue("Unexpected Exception: " + listener.exception, listener.exception instanceof InterruptedException);
412         TestCase.assertEquals("Expected init to be called", 1 , listener.initialised);
413         TestCase.assertEquals("fileNotFound should not be called", 0 , listener.notFound);
414         TestCase.assertEquals("fileRotated should be be called", 1 , listener.rotated);
415     }

451     /** Append some lines to a file */
452     private void write(final File file, final String... lines) throws Exception {
453         FileWriter writer = null;
454         try {
455             writer = new FileWriter(file, true);
456             for (final String line : lines) {
457                 writer.write(line + "\n");
458             }
459         } finally {
460             IOUtils.closeQuietly(writer);
461         }
462     }

464     /** Append a string to a file */
465     private void writeString(final File file, final String ... strings) throws Exception {
466         FileWriter writer = null;
467         try {
468             writer = new FileWriter(file, true);
469             for (final String string : strings) {
470                 writer.write(string);
471             }
472         } finally {
473             IOUtils.closeQuietly(writer);
474         }
