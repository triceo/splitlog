1   package com.github.triceo.splitlog.splitters.exceptions;
2   
3   import java.util.regex.Matcher;
4   
5   class StackTraceParser extends ExceptionLineParser<StackTraceLine> {
6   
7       @Override
8       public StackTraceLine parse(final String line) {
9           final String regex = "^at (" + ExceptionLineParser.JAVA_FQN_REGEX + ")\\((.*)\\)( \\~?\\[(.*)\\])?$";
10          final Matcher m = this.getMatcher(regex, line);
11          if (!m.matches()) {
12              return null;
13          }
14          final String methodName = m.group(1);
15          final String lineIdentifier = m.group(3);
16          final boolean providesJarIdentification = m.group(5) != null;
17          final String[] jarIdentificationParts = providesJarIdentification ? m.group(5).split("\\Q:\\E")
18                  : new String[] {};
19          if (providesJarIdentification && (jarIdentificationParts.length != 2)) {
20              return null;
21          }
22          if (lineIdentifier.equals("Unknown Source")) {
23              if (providesJarIdentification) {
24                  return StackTraceLine.newUnknownMethod(methodName, jarIdentificationParts);
25              } else {
26                  return StackTraceLine.newUnknownMethod(methodName);
27              }
28          } else if (lineIdentifier.equals("Native Method")) {
29              if (providesJarIdentification) {
30                  return StackTraceLine.newNativeMethod(methodName, jarIdentificationParts);
31              } else {
32                  return StackTraceLine.newNativeMethod(methodName);
33              }
34          } else {
35              final String[] parts = lineIdentifier.split("\\Q:\\E");
36              if ((parts.length != 2) || !parts[1].matches("\\d+")) {
37                  // not a line number
38                  return null;
39              }
40              if (providesJarIdentification) {
41                  return StackTraceLine
42                          .newMethod(methodName, parts[0], Integer.parseInt(parts[1]), jarIdentificationParts);
43              } else {
44                  return StackTraceLine.newMethod(methodName, parts[0], Integer.parseInt(parts[1]));
45              }
46          }
47      }

49  }
