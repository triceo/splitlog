6   import java.util.List;
7   import java.util.UUID;

9   import org.apache.commons.io.FileUtils;
10  import org.assertj.core.api.Assertions;
11  import org.junit.Test;
12  import org.slf4j.Logger;
13  import org.slf4j.LoggerFactory;
14  
15  import com.github.triceo.splitlog.api.Follower;
16  import com.github.triceo.splitlog.api.LogWatchBuilder;
17  
18  public abstract class AbstractMessageWritingTest extends DefaultFollowerBaseTest {
19  
20      private static final Logger LOGGER = LoggerFactory.getLogger(AbstractMessageWritingTest.class);
21      private static final int MESSAGES_TO_WRITE = 10;
22  
23      public AbstractMessageWritingTest(final LogWatchBuilder builder) {
24          super(builder);
25      }
26  
27      @Test
28      public abstract void testWriteMessages();
29  
30      protected void writeAndTest(final boolean closeBeforeWriting) {
31          final Follower follower = this.getLogWatch().startFollowing();
32          final List<String> messages = new LinkedList<String>();
33          for (int i = 0; i < AbstractMessageWritingTest.MESSAGES_TO_WRITE; i++) {
34              messages.add(LogWriter.write(follower, UUID.randomUUID().toString()));
35          }
36          messages.remove(messages.size() - 1); // last message will not be
37          // written
38          if (closeBeforeWriting) {
39              this.getLogWatch().stopFollowing(follower);
40          }
41          try {
42              final File f = LogWriter.createTempFile();
43              AbstractMessageWritingTest.LOGGER.info("Will write into '{}'.", f);
44              follower.write(new FileOutputStream(f));
45              Assertions.assertThat(f).exists();
46              final List<String> lines = FileUtils.readLines(f, "UTF-8");
47              Assertions.assertThat(lines).isEqualTo(messages);
48          } catch (final Exception e) {
49              Assertions.fail("Couldn't write to file.");
50          } finally {
51              if (this.getLogWatch().isFollowedBy(follower)) {
52                  this.getLogWatch().stopFollowing(follower);
53              }
54          }
55      }
56  
57  }
