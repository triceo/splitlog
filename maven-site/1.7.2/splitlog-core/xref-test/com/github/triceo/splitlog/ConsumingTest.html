17      private static final class CountingMessageListener<T extends MessageProducer<T>> implements MessageListener<T> {
18  
19          private int numCalls = 0;
20  
21          public int getNumberOfTimesCalled() {
22              return this.numCalls;
23          }
24  
25          @Override
26          public void messageReceived(final Message message, final MessageDeliveryStatus status, final T producer) {
27              if (status != MessageDeliveryStatus.INCOMING) {
28                  return;
29              }
30              this.numCalls++;
31          }
32  
33      }
34  
35      private static final class FailingMessageListener<T extends MessageProducer<T>> implements MessageListener<T> {
36  
37          @Override
38          public void messageReceived(final Message message, final MessageDeliveryStatus status, final T producer) {
39              Assertions.fail("This shouldn't have been executed.");
40          }
41  
42      }

44      @Test
45      public void testCallPropagation1() {
46          final DefaultLogWatch w = (DefaultLogWatch) this.getLogWatch();
47          final CountingMessageListener<LogWatch> l1 = new CountingMessageListener<LogWatch>();
48          // direct consumption
49          w.startConsuming(l1);
50          w.messageIncoming(new MessageBuilder("test").buildIntermediate());
51          Assertions.assertThat(l1.getNumberOfTimesCalled()).isEqualTo(1);
52          // indirection of first order
53          final CountingMessageListener<Follower> l2 = new CountingMessageListener<Follower>();
54          final Follower f = w.startFollowing();
55          f.startConsuming(l2);
56          w.messageIncoming(new MessageBuilder("test2").buildIntermediate());
57          Assertions.assertThat(l2.getNumberOfTimesCalled()).isEqualTo(1);
58          // indirection of second order
59          final CountingMessageListener<MergingFollower> l3 = new CountingMessageListener<MergingFollower>();
60          final Follower f2 = w.startFollowing();
61          final MergingFollower mf = f2.mergeWith(f);
62          mf.startConsuming(l3);
63          w.messageIncoming(new MessageBuilder("test3").buildIntermediate());
64          /*
65           * will receive the same message from two followers
66           */
67          Assertions.assertThat(l3.getNumberOfTimesCalled()).isEqualTo(2);
68          /*
69           * and check the state of all other consumers
70           */
71          Assertions.assertThat(l2.getNumberOfTimesCalled()).isEqualTo(2);
72          Assertions.assertThat(l1.getNumberOfTimesCalled()).isEqualTo(3);
73      }

75      @Test
76      public void testConsumerEquality() {
77          final DefaultLogWatch w = (DefaultLogWatch) this.getLogWatch();
78          final CountingMessageListener<LogWatch> l1 = new CountingMessageListener<LogWatch>();
79          final MessageConsumer<LogWatch> c1 = w.startConsuming(l1);
80          final MessageConsumer<LogWatch> c2 = w.startConsuming(l1);
81          Assertions.assertThat(c2).isSameAs(c1);
82          final CountingMessageListener<LogWatch> l2 = new CountingMessageListener<LogWatch>();
83          final MessageConsumer<LogWatch> c3 = w.startConsuming(l2);
84          final MessageConsumer<LogWatch> c4 = w.startConsuming(l1);
85          Assertions.assertThat(c3).isNotSameAs(c1);
86          Assertions.assertThat(c4).isSameAs(c1);
87      }

124     private MergingFollower testStopPropagationWithMerging() {
125         final LogWatch w = this.getLogWatch();
126         final Follower f = w.startFollowing();
127         final Follower f2 = w.startFollowing();
128         final MergingFollower m = f.mergeWith(f2);
129         final MessageConsumer<MergingFollower> c = m.startConsuming(new FailingMessageListener<MergingFollower>());
130         // main propagation
131         Assertions.assertThat(m.stop()).isTrue();
132         Assertions.assertThat(c.isStopped()).isTrue();
133         // side effects
134         Assertions.assertThat(f.isStopped()).isTrue();
135         Assertions.assertThat(f.stop()).isFalse();
136         Assertions.assertThat(f2.isStopped()).isTrue();
137         Assertions.assertThat(f2.stop()).isFalse();
138         return m;
139     }
