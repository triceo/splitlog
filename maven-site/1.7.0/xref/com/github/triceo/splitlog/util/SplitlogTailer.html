3   import java.io.File;
4   import java.util.concurrent.TimeUnit;
5   
6   import org.apache.commons.io.input.Tailer;
7   import org.apache.commons.io.input.TailerListener;
8   import org.slf4j.Logger;
9   
10  import com.github.triceo.splitlog.logging.SplitlogLoggerFactory;
11  
12  /**
13   * Extension of the Apache tailer to allow us to monitor when it's actually
14   * started tailing.
15   */
16  public class SplitlogTailer extends Tailer {
17  
18      private static final Logger LOGGER = SplitlogLoggerFactory.getLogger(SplitlogTailer.class);
19  
20      public SplitlogTailer(final File file, final TailerListener listener, final long delayMillis, final boolean end,
21          final boolean reOpen, final int bufSize) {
22          super(file, listener, delayMillis, end, reOpen, bufSize);
23      }
24  
25      @Override
26      public void run() {
27          SplitlogTailer.LOGGER.info("Tailer thread started.");
28          final long start = System.nanoTime();
29          super.run();
30          final long result = System.nanoTime() - start;
31          SplitlogTailer.LOGGER.info("Tailer thread stopped, took {} ms.", TimeUnit.NANOSECONDS.toMillis(result));
32      }
33  
34  }
