6   import org.assertj.core.api.Assertions;

11  import com.github.triceo.splitlog.api.Follower;
12  import com.github.triceo.splitlog.api.LogWatchBuilder;
13  import com.github.triceo.splitlog.api.Message;
14  import com.github.triceo.splitlog.api.SimpleMessageCondition;
15  
16  @RunWith(Parameterized.class)
17  public class TaggingTest extends DefaultFollowerBaseTest {
18  
19      private static final class NothingAcceptingMessageCondition implements SimpleMessageCondition {
20  
21          @Override
22          public boolean accept(final Message evaluate) {
23              return false;
24          }
25  
26      }
27  
28      public TaggingTest(final LogWatchBuilder builder) {
29          super(builder);
30      }
31  
32      @Test
33      public void testTag() {
34          final String message1 = "test";
35          final String message2 = "test7";
36          final String message3 = "test11";
37          final String tag0 = "tag1";
38          final String tag1 = "tag1";
39          final String tag2 = "tag2";
40          final Follower follower = this.getLogWatch().startFollowing();
41          follower.tag(tag0);
42          String result = LogWriter.write(follower, message1);
43          Assertions.assertThat(result).isEqualTo(message1);
44          /*
45           * when this tag is being created, the first message is already
46           * instantiated. therefore, it will come after this message.
47           */
48          follower.tag(tag1);
49          result = LogWriter.write(follower, message2);
50          Assertions.assertThat(result).isEqualTo(message2);
51          // ditto
52          follower.tag(tag2);
53          result = LogWriter.write(follower, message3);
54          Assertions.assertThat(result).isEqualTo(message3);
55          // make sure all messages are present with the default condition
56          SortedSet<Message> messages = new TreeSet<Message>(follower.getMessages());
57          DefaultFollowerBaseTest.assertProperOrder(messages, tag0, message1, tag1, message2, tag2);
58          // make sure just the tags are present witn nothing-accepting condition
59          messages = new TreeSet<Message>(follower.getMessages(new NothingAcceptingMessageCondition()));
60          DefaultFollowerBaseTest.assertProperOrder(messages, tag0, tag1, tag2);
61      }
