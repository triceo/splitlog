11  import com.github.triceo.splitlog.api.Follower;
12  import com.github.triceo.splitlog.api.LogWatchBuilder;
13  import com.github.triceo.splitlog.api.Message;
14  
15  @RunWith(Parameterized.class)
16  public class NestingTest extends DefaultFollowerBaseTest {
17  
18      public NestingTest(final LogWatchBuilder builder) {
19          super(builder);
20      }
21  
22      @Test
23      public void testNesting() {
24          final String message1 = "test1";
25          final String message2 = "test2";
26          final String message3 = "test3";
27          final String message4 = "test4";
28          final String message5 = "test5";
29          final Follower follower = this.getLogWatch().startFollowing();
30          // make sure the messages are received by the first follower
31          LogWriter.write(follower, message1);
32          String result = LogWriter.write(follower, message2);
33          Assertions.assertThat(result).isEqualTo(message2);
34          Assertions.assertThat(follower.getMessages().size()).isEqualTo(1);
35          // start a second follower, send some messages
36          final Follower nestedFollower = this.getLogWatch().startFollowing();
37          result = LogWriter.write(follower, message3);
38          result = LogWriter.write(follower, message4);
39          Assertions.assertThat(result).isEqualTo(message4);
40          this.getLogWatch().stopFollowing(nestedFollower);
41          // send another message, so the original follower has something extra
42          Assertions.assertThat(nestedFollower.isStopped()).isTrue();
43          result = LogWriter.write(follower, message5);
44          // and make sure that the original follower has all messages
45          final SortedSet<Message> messages = new TreeSet<Message>(follower.getMessages());
46          DefaultFollowerBaseTest.assertProperOrder(messages, message1, message2, message3, message4);
47          // and the nested follower has only the two while it was running
48          final SortedSet<Message> nestedMessages = new TreeSet<Message>(nestedFollower.getMessages());
49          DefaultFollowerBaseTest.assertProperOrder(nestedMessages, message2, message3);
50      }
