40  /**
41   * Base class for testcases doing tests with files.
42   */
43  public abstract class FileBasedTestCase extends TestCase {
44  
45      private static volatile File testDir;
46  
47      public static File getTestDirectory() {
48          if (FileBasedTestCase.testDir == null) {
49              FileBasedTestCase.testDir = new File("test/io/").getAbsoluteFile();
50          }
51          FileBasedTestCase.testDir.mkdirs();
52          return FileBasedTestCase.testDir;
53      }
54  
55      public FileBasedTestCase(final String name) {
56          super(name);
57      }
58  
59      /** Assert that the content of a file is equal to that in a byte[]. */
60      protected void assertEqualContent(final byte[] b0, final File file) throws IOException {
61          final InputStream is = new java.io.FileInputStream(file);
62          int count = 0, numRead = 0;
63          final byte[] b1 = new byte[b0.length];
64          try {
65              while ((count < b0.length) && (numRead >= 0)) {
66                  numRead = is.read(b1, count, b0.length);
67                  count += numRead;
68              }
69              TestCase.assertEquals("Different number of bytes: ", b0.length, count);
70              for (int i = 0; i < count; i++) {
71                  TestCase.assertEquals("byte " + i + " differs", b0[i], b1[i]);
72              }
73          } finally {
74              is.close();
75          }
76      }
77  
78      /** Assert that the content of a file is equal to that in a char[]. */
79      protected void assertEqualContent(final char[] c0, final File file) throws IOException {
80          final Reader ir = new java.io.FileReader(file);
81          int count = 0, numRead = 0;
82          final char[] c1 = new char[c0.length];
83          try {
84              while ((count < c0.length) && (numRead >= 0)) {
85                  numRead = ir.read(c1, count, c0.length);
86                  count += numRead;
87              }
88              TestCase.assertEquals("Different number of chars: ", c0.length, count);
89              for (int i = 0; i < count; i++) {
90                  TestCase.assertEquals("char " + i + " differs", c0[i], c1[i]);
91              }
92          } finally {
93              ir.close();
94          }
95      }
96  
97      /** Assert that the content of two files is the same. */
98      private void assertEqualContent( final File f0, final File f1 )
99              throws IOException
100     {
101         /* This doesn't work because the filesize isn't updated until the file
102          * is closed.
103         assertTrue( "The files " + f0 + " and " + f1 +
104                     " have differing file sizes (" + f0.length() +
105                     " vs " + f1.length() + ")", ( f0.length() == f1.length() ) );
106          */
107         final InputStream is0 = new java.io.FileInputStream( f0 );
108         try {
109             final InputStream is1 = new java.io.FileInputStream( f1 );
110             try {
111                 final byte[] buf0 = new byte[ 1024 ];
112                 final byte[] buf1 = new byte[ 1024 ];
113                 int n0 = 0;
114                 int n1 = 0;
115 
116                 while( -1 != n0 )
117                 {
118                     n0 = is0.read( buf0 );
119                     n1 = is1.read( buf1 );
120                     TestCase.assertTrue( "The files " + f0 + " and " + f1 +
121                             " have differing number of bytes available (" + n0 +
122                             " vs " + n1 + ")", ( n0 == n1 ) );
123 
124                     TestCase.assertTrue( "The files " + f0 + " and " + f1 +
125                             " have different content", Arrays.equals( buf0, buf1 ) );
126                 }
127             } finally {
128                 is1.close();
129             }
130         } finally {
131             is0.close();
132         }
133     }
134 
135     protected void checkFile( final File file, final File referenceFile )
136             throws Exception {
137         TestCase.assertTrue( "Check existence of output file", file.exists() );
138         this.assertEqualContent( referenceFile, file );
139     }
140 
141     protected void checkWrite(final OutputStream output) throws Exception {
142         try {
143             new java.io.PrintStream(output).write(0);
144         } catch (final Throwable t) {
145             throw new AssertionFailedError(
146                     "The copy() method closed the stream "
147                             + "when it shouldn't have. "
148                             + t.getMessage());
149         }
150     }
151 
152     protected void checkWrite(final Writer output) throws Exception {
153         try {
154             new java.io.PrintWriter(output).write('a');
155         } catch (final Throwable t) {
156             throw new AssertionFailedError(
157                     "The copy() method closed the stream "
158                             + "when it shouldn't have. "
159                             + t.getMessage());
160         }
161     }
162 
163     protected void createFile(final File file, final long size)
164             throws IOException {
165         if (!file.getParentFile().exists()) {
166             throw new IOException("Cannot create file " + file
167                     + " as the parent directory does not exist");
168         }
169         final BufferedOutputStream output =
170                 new BufferedOutputStream(new java.io.FileOutputStream(file));
171         try {
172             this.generateTestData(output, size);
173         } finally {
174             IOUtils.closeQuietly(output);
175         }
176     }
177 
178     protected void createLineBasedFile(final File file, final String[] data) throws IOException {
179         if ((file.getParentFile() != null) && !file.getParentFile().exists()) {
180             throw new IOException("Cannot create file " + file + " as the parent directory does not exist");
181         }
182         final PrintWriter output = new PrintWriter(new OutputStreamWriter(new FileOutputStream(file), "UTF-8"));
183         try {
184             for (final String element : data) {
185                 output.println(element);
186             }
187         } finally {
188             IOUtils.closeQuietly(output);
189         }
190     }
191 
192     protected void deleteFile( final File file )
193             throws Exception {
194         if (file.exists()) {
195             TestCase.assertTrue("Couldn't delete file: " + file, file.delete());
196         }
197     }
198 
199     protected byte[] generateTestData(final long size) {
200         try {
201             final ByteArrayOutputStream baout = new ByteArrayOutputStream();
202             this.generateTestData(baout, size);
203             return baout.toByteArray();
204         } catch (final IOException ioe) {
205             throw new RuntimeException("This should never happen: " + ioe.getMessage());
206         }
207     }
208 
209     protected void generateTestData(final OutputStream out, final long size)
210             throws IOException {
211         for (int i = 0; i < size; i++) {
212             //output.write((byte)'X');
213 
214             // nice varied byte pattern compatible with Readers and Writers
215             out.write( (byte)( (i % 127) + 1) );
216         }
217     }
218 
219     protected File newFile(final String filename) throws IOException {
220         final File destination = new File( FileBasedTestCase.getTestDirectory(), filename );
221         /*
222         assertTrue( filename + "Test output data file shouldn't previously exist",
223                     !destination.exists() );
224          */
225         if (destination.exists()) {
226             FileUtils.forceDelete(destination);
227         }
228         return destination;
229     }
